CREATE TABLE IF NOT EXISTS Besitzer(PersonID int NOT NULL,Vorname varchar(255) NOT NULL,Nachname varchar(255) NOT NULL,Geburtsdatum varchar(255) NOT NULL,Gehalt int NOT NULL,Land varchar(3) NOT NULL,Ort varchar(255) NOT NULL,Adresse varchar(255) NOT NULL,PRIMARY KEY (PersonID));
CREATE TABLE IF NOT EXISTS Hersteller(HerstellerID int NOT NULL,Name varchar(255) NOT NULL,Gruendungsjahr varchar(4) NOT NULL,PRIMARY KEY (HerstellerID));
CREATE TABLE IF NOT EXISTS Auto(PersonID int NOT NULL,HerstellerID int NOT NULL,Marke varchar(255) NOT NULL,Baujahr varchar(4) NOT NULL,Motorleistung int NOT NULL,FOREIGN KEY (PersonID) REFERENCES Besitzer(PersonID),FOREIGN KEY (HerstellerID) REFERENCES Hersteller(HerstellerID));
